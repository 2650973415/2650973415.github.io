Java连接Mysql

这里用IntelliJ IDEA添加驱动！！

Mysql安装就不介绍了。接下来就是下载Mysql驱动包“mysql-connector-java-5.1.27.jar”，也不介绍了，百度都有！准备好这些就可以开始了：

  • 加载驱动 在项目目录下创建lib目录，然后把下载好的mysql-connector-java-5.1.27.jar放到该目录下，如图所示：

    然后右击驱动包选择Add as Library，然后点OK 如图所示：

    接下来就是java连接到数据库了

  • java连接到数据库并查询数据

import java.sql.*;

public class Test_Mysql {
        public static void main(String[] args) {
            String url = "jdbc:mysql://localhost:3306/zhao";//要访问的数据库地址
            String user = "root";//用户名
            String password = "1002";//密码
            //创建Connection对象
            Connection connection = null;
            try {
                //加载Mysql驱动程序
                Class.forName("com.mysql.jdbc.Driver");
                //连接到数据库
                connection = DriverManager.getConnection(url, user, password);
                //创建Statement对象用来执行sql语句
                Statement statement = connection.createStatement();
                //要执行的SQL语句
                String sql = "select * from students";
                //ResultSet类，用来存放获取的结果集！！
                ResultSet resultSet = statement.executeQuery(sql);
                //while遍历结果集
                while (resultSet.next()){
                    //获取字段数据
                    int sid = resultSet.getInt("id");//id字段数据
                    String sname = resultSet.getString("name");//name字段数据
                    String ssex = resultSet.getString("sex");//sex字段数据
                    int sage = resultSet.getInt("age");//age字段数据
                    System.out.println(sid+"--"+sname+"--"+ssex+"--"+sage);
                }

            } catch (Exception e) {
                e.printStackTrace();
            }
        }
}

  • 更新数据
//将name为小明的age更新为17
PreparedStatement preparedStatement = null;
try {
    preparedStatement = connection.prepareStatement("UPDATE students SET age = ? WHERE name = ?");
    //其中有两个参数--“？”
    preparedStatement.setInt(1,17);//设置参数1
    preparedStatement.setString(2,"小明");//设置参数2
    preparedStatement.executeUpdate();//执行更新
} catch (SQLException e) {
    e.printStackTrace();
}
  • 添加数据
//添加一条数据id=10，name=‘小明’，sex=‘男’，age=18
PreparedStatement preparedStatement = null;
try {
    preparedStatement = connection.prepareStatement("INSERT INTO students VALUES(?,?,?,?)");
    preparedStatement.setInt(1,10);//设置参数1也就是表的第一个字段id
    preparedStatement.setString(2,"小明");//设置参数2（name）
    preparedStatement.setString(3,"男");//设置参数3（sex）
    preparedStatement.setInt(4,18);//设置参数4（age）
    preparedStatement.executeUpdate();//执行更新
    System.out.println("添加成功");
} catch (SQLException e) {
    e.printStackTrace();
}

//删除id为1的数据
PreparedStatement preparedStatement = null;
